\section*{Parameters}
\[
\begin{alignedat}{3}
&a_{i,t} &&[\mathrm{USD}/\mathrm{kW}] &&\text{inverse-demand intercept} \\
&b_{i,t} &&[\mathrm{USD}/(\mathrm{kW}\cdot\mathrm{GW})] &&\text{inverse-demand slope} \\
&D^{\max}_{i,t} &&[\mathrm{GW}] &&\text{demand cap} \\
&K_i^0 &&[\mathrm{GW}] &&\text{initial capacity} \\
&c^{man}_{i,t} &&[\mathrm{USD}/\mathrm{kW}] &&\text{manufacturing cost schedule} \\
&c^{ship}_{e,i} &&[\mathrm{USD}/\mathrm{kW}] &&\text{shipping cost} \\
&\bar p_{e,i} &&[\mathrm{USD}/\mathrm{kW}] &&\text{offer-price upper bound} \\
&g_i^{exp} &&[\mathrm{GW/yr}\ \text{or}\ 1/\mathrm{yr}] &&\text{capacity expansion limit} \\
&g_i^{dec} &&[1/\mathrm{yr}] &&\text{capacity decommission limit} \\
&f_i^{hold} &&[\mathrm{mUSD}/(\mathrm{GW}\cdot\mathrm{yr})] &&\text{capacity holding cost} \\
&c_i^{inv} &&[\mathrm{mUSD}/(\mathrm{GW}\cdot\mathrm{yr})] &&\text{investment cost} \\
&\beta_t &&[-] &&\text{discount factor} \\
&y_t &&[\mathrm{yr}] &&\text{years represented by block } t \\
&\epsilon_x &&[\mathrm{USD}/(\mathrm{kW}\cdot\mathrm{GW})] &&\text{trade regularization} \\
&\epsilon_{comp} &&[-] &&\text{complementarity relaxation} \\
&c_q^{pen},\,c_p^{pen},\,c_a^{pen} &&[-] &&\text{proximal penalty weights} \\
&c_q^{quad},\,c_p^{quad},\,c_a^{quad} &&[-] &&\text{economic quadratic penalty weights} \\
&\rho^{keep},\,\rho^{sub},\,\rho^{term},\,\rho^{dec} &&[-] &&\text{capacity-policy coefficients}
\end{alignedat}
\]

\section*{Sets}
\[
\mathcal{R}:\text{ regions},\qquad
\mathcal{P}\subseteq\mathcal{R}:\text{ strategic players},\qquad
\mathcal{T}:\text{ time periods},\qquad
\mathcal{T}^{-}=\mathcal{T}\setminus\{t^*\},\qquad
t_0=\min\mathcal{T},\qquad
t^*=\max\mathcal{T}.
\]
\[
e\in\mathcal{R}\ \text{(exporter)},\qquad
i,j\in\mathcal{R}\ \text{(importer / market)},\qquad
t\in\mathcal{T},\qquad
r\in\mathcal{P}\ \text{(active player)}.
\]

\section*{LLP Primal}
\begin{align}
\min_{x,x^{dem}} \quad
z^{LLP}
= {} &
\sum_{t\in\mathcal{T}}\sum_{e\in\mathcal{R}}\sum_{i\in\mathcal{R}}
\left(p_{e,i,t}+c^{ship}_{e,i}\right)x_{e,i,t}
+ \sum_{t\in\mathcal{T}}\sum_{e\in\mathcal{R}}\sum_{i\in\mathcal{R}}
\frac{\epsilon_x}{2}x_{e,i,t}^2
\nonumber\\
&-
\sum_{t\in\mathcal{T}}\sum_{i\in\mathcal{R}}
\left(
a^{bid}_{i,t}x^{dem}_{i,t}
- \frac{b_{i,t}}{2}(x^{dem}_{i,t})^2
\right)
\end{align}
\begin{align}
\sum_{e\in\mathcal{R}} x_{e,i,t} - x^{dem}_{i,t} &= 0
&& \forall i,t
&& \{\lambda_{i,t}\}
\label{eq:llp-bal}
\\
Q_{e,t} - \sum_{i\in\mathcal{R}} x_{e,i,t} &\ge 0
&& \forall e,t
&& \{\mu^{offer}_{e,t}\ge 0\}
\label{eq:llp-cap}
\\
D^{\max}_{i,t} - x^{dem}_{i,t} &\ge 0
&& \forall i,t
&& \{\beta^{dem}_{i,t}\ge 0\}
\label{eq:llp-dem-up}
\\
x^{dem}_{i,t} &\ge 0
&& \forall i,t
&& \{\psi_{i,t}\ge 0\}
\label{eq:llp-dem-lo}
\\
x_{e,i,t} &\ge 0
&& \forall e,i,t
&& \{\gamma_{e,i,t}\ge 0\}.
\label{eq:llp-x-lo}
\end{align}

\section*{LLP KKTs}
\begin{align}
\frac{\partial \mathcal{L}^{LLP}}{\partial x_{e,i,t}}:\quad
\left(p_{e,i,t}+c^{ship}_{e,i}\right)
+ \epsilon_x x_{e,i,t}
- \lambda_{i,t}
+ \mu^{offer}_{e,t}
- \gamma_{e,i,t}
&= 0
&& \forall e,i,t
\\
\frac{\partial \mathcal{L}^{LLP}}{\partial x^{dem}_{i,t}}:\quad
-a^{bid}_{i,t}
+ b_{i,t}x^{dem}_{i,t}
+ \lambda_{i,t}
+ \beta^{dem}_{i,t}
- \psi_{i,t}
&= 0
&& \forall i,t
\end{align}
\begin{align}
\mu^{offer}_{e,t} &\ge 0
&& \forall e,t
\\
\gamma_{e,i,t} &\ge 0
&& \forall e,i,t
\\
\beta^{dem}_{i,t} &\ge 0
&& \forall i,t
\\
\psi_{i,t} &\ge 0
&& \forall i,t
\\
\lambda_{i,t} &\in \mathbb{R}
&& \forall i,t
\end{align}
\begin{align}
0 \le \mu^{offer}_{e,t} \perp
\left(Q_{e,t} - \sum_{i\in\mathcal{R}} x_{e,i,t}\right) &\ge 0
&& \forall e,t
\\
0 \le \gamma_{e,i,t} \perp x_{e,i,t} &\ge 0
&& \forall e,i,t
\\
0 \le \beta^{dem}_{i,t} \perp
\left(D^{\max}_{i,t} - x^{dem}_{i,t}\right) &\ge 0
&& \forall i,t
\\
0 \le \psi_{i,t} \perp x^{dem}_{i,t} &\ge 0
&& \forall i,t
\end{align}
\[
\text{implemented relaxation:}\qquad
\mu^{offer}_{e,t}\!\left(Q_{e,t}-\sum_i x_{e,i,t}\right)\le \epsilon_{comp},\quad
\gamma_{e,i,t}x_{e,i,t}\le \epsilon_{comp},\quad
\beta^{dem}_{i,t}\!\left(D^{\max}_{i,t}-x^{dem}_{i,t}\right)\le \epsilon_{comp},\quad
\psi_{i,t}x^{dem}_{i,t}\le \epsilon_{comp}.
\]

\section*{ULP Primal}
\begin{align}
\max \quad \Pi_r
= {} &
\sum_{t\in\mathcal{T}} \beta_t y_t
\Bigg[
\left(
a_{r,t}x^{dem}_{r,t}
- \frac{b_{r,t}}{2}(x^{dem}_{r,t})^2
- \lambda_{r,t}x^{dem}_{r,t}
\right)
\nonumber\\
&\qquad\qquad
+ \sum_{j\in\mathcal{R}}
\left(
\lambda_{j,t}
- c^{man}_{r,t}
- c^{ship}_{r,j}
\right)x_{r,j,t}
- f^{hold}_{r}K_{r,t}
- c^{inv}_{r}I^+_{r,t}
\nonumber\\
&\qquad\qquad
+ \rho^{keep}K_{r,t}
+ \rho^{sub}I^+_{r,t}
- \rho^{dec}D^-_{r,t}
- \frac{c_q^{quad}}{2}\left(Q_{r,t}-K_{r,t}\right)^2
\nonumber\\
&\qquad\qquad
- \frac{c_p^{quad}}{2}\sum_{j\in\mathcal{R}}
\left(p_{r,j,t}-c^{man}_{r,t}\right)^2
- \frac{c_a^{quad}}{2}\left(a_{r,t}-a^{bid}_{r,t}\right)^2
\nonumber\\
&\qquad\qquad
- \frac{c_p^{pen}}{2}\sum_{j\in\mathcal{R}}
\left(p_{r,j,t}-p^{last}_{r,j,t}\right)^2
- \frac{c_q^{pen}}{2}\left(Q_{r,t}-Q^{last}_{r,t}\right)^2
- \frac{c_a^{pen}}{2}\left(a^{bid}_{r,t}-a^{last}_{r,t}\right)^2
\Bigg]
\nonumber\\
&\quad
+ \beta_{t^*}\rho^{term}c^{inv}_{r}\sum_{t\in\mathcal{T}} y_t I^+_{r,t}
\end{align}
\begin{align}
K_{i,t_0} &= K_i^0
&& \forall i
\\
K_{i,t^+} &= K_{i,t} + y_t\left(I^+_{i,t}-D^-_{i,t}\right)
&& \forall i,\ t\in\mathcal{T}^{-}
\\
0 \le I^+_{i,t} &\le g_i^{exp}
&& \forall i,\ t\in\mathcal{T}^{-}
\qquad \text{if } g_i^{exp}\text{ is absolute}
\\
0 \le I^+_{i,t} &\le g_i^{exp}K_{i,t}
&& \forall i,\ t\in\mathcal{T}^{-}
\qquad \text{if } g_i^{exp}\text{ is proportional}
\\
0 \le D^-_{i,t} &\le g_i^{dec}K_{i,t}
&& \forall i,\ t\in\mathcal{T}^{-}
\\
I^+_{i,t^*} = D^-_{i,t^*} &= 0
&& \forall i
\\
0 \le Q_{i,t} &\le K_{i,t}
&& \forall i,t
\qquad \text{or } Q_{i,t}=K_{i,t}\ \text{if quantity withholding is disabled}
\\
p_{i,i,t} &= c^{man}_{i,t}
&& \forall i,t
\\
0 \le p_{i,j,t} &\le \bar p_{i,j}
&& \forall i,j,t
\\
0 \le a^{bid}_{i,t} &\le a_{i,t}
&& \forall i,t
\qquad \text{or } a^{bid}_{i,t}=a_{i,t}\ \text{if demand withholding is disabled}
\\
K_{i,t},\,x_{e,i,t},\,x^{dem}_{i,t},\,Q_{i,t} &\ge 0
&& \forall i,e,t
\end{align}
\[
\text{s.t. LLP KKT conditions above.}
\]
