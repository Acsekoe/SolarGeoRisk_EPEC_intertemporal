\section*{Non-Strategic Primal LLP (Unconstrained Capacity, No Tariffs)}

\subsection*{Variables}
\begin{itemize}
    \item $x_e^{man} \ge 0$: Quantity manufactured in region $e$ (unconstrained from above)
    \item $x_{ei} \ge 0$: Trade flow from $e$ to $i$
    \item $0 \le x_i^{dem} \le D_i^{\max}$: Quantity consumed in region $i$
\end{itemize}

\subsection*{Objective: Minimize Cost minus Utility}
\begin{align}
\min_{x,\,x^{man},\,x^{dem}} \quad
\mathcal{Z}_{LL}
&=
\sum_{t \in T} \beta_t y_t \Bigg[
\sum_{e \in R} c_e^{man}\,x_e^{man}
+
\sum_{e \in R}\sum_{i \in R}
c_{ei}^{ship}\,x_{ei}
\nonumber\\
&\quad
-
\sum_{i \in R}
\left(
a_i x_i^{dem} - \frac{1}{2} b_i (x_i^{dem})^2
\right)
\Bigg]
\end{align}

\subsection*{Constraints}

\subsubsection*{Material Balance (Production equals Exports)}
\begin{equation}
x_e^{man} - \sum_{i \in R} x_{ei} = 0 \qquad \forall e \in R,\; t \in T
\end{equation}

\subsubsection*{Demand Balance (Imports equal Consumption)}
\begin{equation}
\sum_{e \in R} x_{ei} - x_i^{dem} = 0 \qquad \forall i \in R,\; t \in T
\end{equation}

\subsubsection*{Non-negativity and Demand Bound}
\begin{align}
x_e^{man} &\ge 0 && \forall e \in R,\; t \in T \\
x_i^{dem} &\in [0,\, D_i^{\max}] && \forall i \in R,\; t \in T \\
x_{ei} &\ge 0 && \forall e, i \in R,\; t \in T
\end{align}

\subsubsection*{Equilibrium Prices}
The dual $\lambda_i^t$ of the demand balance equation is the market clearing price.
With China as the lowest-cost supplier, prices settle approximately at:
\begin{equation}
\lambda_i^t \approx c_{CH}^{man} + c_{CH,i}^{ship}
\end{equation}
